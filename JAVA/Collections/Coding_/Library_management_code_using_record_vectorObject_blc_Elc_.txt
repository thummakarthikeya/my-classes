package Blc;

import java.util.Vector;
import java.util.*;

public class Library {
	
	
	public void  displayAvailableBooks( Vector<Book> books) {
		for(Book book : books) {
			IO.println(book);
		}
	}

	
	public void issueBook(Vector<Book> books, Scanner sc) {
		IO.println("Enter the book id which u want : ");
		int bookid = sc.nextInt();
		boolean flag = false;
		for(int i=0; i<books.size(); i++) {
			if(books.get(i).id() == bookid){
				IO.println(books.get(i));
				books.remove(i);
				flag = true;	
			}
		}
		if(!flag) {
			IO.println("Your booke"+bookid+" is not available...");
		}
		
	}
	
	public void searchBook(Vector<Book> books, Scanner sc) {
		IO.println("Enter the book id which book u want to search ");
		int bookid = sc.nextInt();
		for(int i=0; i<books.size(); i++) {
			if(books.get(i).id() == bookid) {
				IO.println(books.get(i));
			}
		}
	}
	
}
